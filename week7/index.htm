  1. Init-Time Branching
    1. This is used to check if a feature is supported and if it is not then the function is changed to the supported version and when it is called again the browser does not need to check if the feature is supported and quickly runs the supported version. This is done with an if statement

function ride(){

    if (window.unicorn) {

        ride = function(){

        // some code that uses the brand new and sparkly unicorn methods

        return 'Riding on a unicorn is the best!';

        }

    } else {

        ride = function(){

        // some code that uses the older pony methods

        return 'Riding on a pony is still pretty good';

        }

    }

    return ride();

}

AJAX

Ajax is a technique that allows web pages to communicate asynchronously with a server, and it dynamically updates web pages without reloading. This enables data to be sent and received in the background, as well as portions of a page to be updated in response to user events, while the rest of the program continues to run.

const form = document.forms['todo'];

form.addEventListener('submit', addTask, false);

 

function addTask(event) {

    event.preventDefault();

    const number = form.task.value;

    const task = {

        userId: 1,

        title: form.task.value,

        completed: false

    }

    const data = JSON.stringify(task);

    const url = 'https://jsonplaceholder.typicode.com/todos';

 

    const headers = new Headers({

        'Accept': 'application/json',

        'Content-Type': 'application/json'

    });

    const request = new Request(url,

    {

        method: 'POST',

        header: headers,

        body: data

    }

    )

 

    fetch(request)

    .then( response => response.json() )

    .then( task => console.log(`Task saved with an id of ${task.id}`) )

    .catch( error => console.log('There was an error:', error))

 

}

        mode: 'cors',
